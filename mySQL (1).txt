CREATE TABLE MARCA (
  idMarca INT AUTO_INCREMENT PRIMARY KEY,
  nombreMarca VARCHAR(100) NOT NULL,
  paisOrigen VARCHAR(300) NOT NULL,
  estado VARCHAR(100) NOT NULL
);

CREATE TABLE TIENDA (
  idTienda INT AUTO_INCREMENT PRIMARY KEY,
  nit INT,
  nombre VARCHAR(100),
  direccion VARCHAR(100),
  telefono VARCHAR(20),
  estado VARCHAR(20)
);

CREATE TABLE PROVEEDOR (
  idProveedor INT AUTO_INCREMENT PRIMARY KEY,
  nit VARCHAR(20) NOT NULL,
  razonSocial VARCHAR(100) NOT NULL,
  contacto VARCHAR(100),
  estado VARCHAR(20) NOT NULL
);

CREATE TABLE CATEGORIA (
  idCategoria INT AUTO_INCREMENT PRIMARY KEY,
  nombreCat VARCHAR(20) NOT NULL,
  descripcion VARCHAR(100),
  estado VARCHAR(100) NOT NULL
);

CREATE TABLE EMPLEADO (
  idEmpleado INT AUTO_INCREMENT PRIMARY KEY,
  sucursal_id INT,
  cargo_id INT,
  num_documento VARCHAR(20) NOT NULL,
  tipo_documento VARCHAR(20) NOT NULL,
  nombres VARCHAR(100) NOT NULL,
  apellidos VARCHAR(100) NOT NULL,
  email VARCHAR(100) NOT NULL,
  telefono VARCHAR(20),
  fecha_ingreso DATE NOT NULL,
  estado VARCHAR(20) NOT NULL
);

CREATE TABLE CLIENTE (
  idCliente INT AUTO_INCREMENT PRIMARY KEY,
  codigoVendedor VARCHAR(20),
  nombre VARCHAR(100),
  cargo VARCHAR(100),
  documento VARCHAR(100),
  telefono VARCHAR(20),
  estado VARCHAR(20)
);

CREATE TABLE PC (
  idPC BIGINT AUTO_INCREMENT PRIMARY KEY,
  marca VARCHAR(100),
  modelo VARCHAR(100),
  precio DOUBLE,
  tipo VARCHAR(200),
  estado VARCHAR(20),
  TIENDAidTienda INT,
  MARCAidMarca INT,
  PROVEEDORidProveedor INT,
  CATEGORIAidCategoria INT,
  CONSTRAINT fk_pc_tienda FOREIGN KEY (TIENDAidTienda) REFERENCES TIENDA(idTienda),
  CONSTRAINT fk_pc_marca FOREIGN KEY (MARCAidMarca) REFERENCES MARCA(idMarca),
  CONSTRAINT fk_pc_proveedor FOREIGN KEY (PROVEEDORidProveedor) REFERENCES PROVEEDOR(idProveedor),
  CONSTRAINT fk_pc_categoria FOREIGN KEY (CATEGORIAidCategoria) REFERENCES CATEGORIA(idCategoria)
);

CREATE TABLE PERIFERICO (
  IDPeriferico INT AUTO_INCREMENT PRIMARY KEY,
  nombre VARCHAR(100) NOT NULL,
  precio DOUBLE NOT NULL,
  estado VARCHAR(20) NOT NULL,
  esGamer INT,
  PCidPC BIGINT,
  MARCAidMarca INT,
  CONSTRAINT fk_peri_pc FOREIGN KEY (PCidPC) REFERENCES PC(idPC),
  CONSTRAINT fk_peri_marca FOREIGN KEY (MARCAidMarca) REFERENCES MARCA(idMarca)
);

CREATE TABLE GARANTIA (
  idGarantia INT AUTO_INCREMENT PRIMARY KEY,
  idPC INT,
  producto_id INT,
  cliente_id INT,
  fecha_inicio DATE NOT NULL,
  numero_serie VARCHAR(100) NOT NULL,
  fecha_fin DATE,
  estado VARCHAR(20) NOT NULL,
  PCidPC BIGINT,
  CONSTRAINT fk_gar_pc FOREIGN KEY (PCidPC) REFERENCES PC(idPC)
);

CREATE TABLE VENTA (
  idVenta INT AUTO_INCREMENT PRIMARY KEY,
  fecha DATE NOT NULL,
  total DOUBLE NOT NULL,
  estado VARCHAR(20) NOT NULL,
  EMPLEADOidEmpleado INT,
  CLIENTEidCliente INT,
  CONSTRAINT fk_venta_empleado FOREIGN KEY (EMPLEADOidEmpleado) REFERENCES EMPLEADO(idEmpleado),
  CONSTRAINT fk_venta_cliente FOREIGN KEY (CLIENTEidCliente) REFERENCES CLIENTE(idCliente)
);

CREATE TABLE DETALLE_VENTA (
  idDetalle INT AUTO_INCREMENT PRIMARY KEY,
  cantidad INT NOT NULL,
  subtotal DECIMAL(12,2) NOT NULL,
  descuento DECIMAL(12,2) DEFAULT 0,
  IVA DECIMAL(12,2) NOT NULL,
  total DECIMAL(12,2) NOT NULL,
  estado VARCHAR(20) NOT NULL,
  PCidPC BIGINT,
  VENTAidVenta INT,
  IDPeriferico INT,
  CONSTRAINT fk_det_pc FOREIGN KEY (PCidPC) REFERENCES PC(idPC),
  CONSTRAINT fk_det_venta FOREIGN KEY (VENTAidVenta) REFERENCES VENTA(idVenta),
  CONSTRAINT fk_det_peri FOREIGN KEY (IDPeriferico) REFERENCES PERIFERICO(IDPeriferico)
);
